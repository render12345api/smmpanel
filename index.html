function doPost(e) {
  try {
    var ss    = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('Orders');
    if (!sheet) {
      sheet = ss.insertSheet('Orders');
      sheet.appendRow([
        'Order ID','Date & Time','Service',
        'Instagram ID / Link','Quantity',
        'Amount (Rs)','Promo Code','Status'
      ]);
      var hdr = sheet.getRange(1,1,1,8);
      hdr.setFontWeight('bold');
      hdr.setBackground('#f09433');
      hdr.setFontColor('#ffffff');
    }
    var data = JSON.parse(e.postData.contents);
    sheet.appendRow([
      data.orderId, data.timestamp, data.service,
      data.target,  data.quantity,  data.amount,
      data.promo,   data.status
    ]);
    return ContentService
      .createTextOutput(JSON.stringify({result:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({result:'error',msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
