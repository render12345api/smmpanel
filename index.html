function doPost(e) {
  try {
    var ss    = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('Orders');

    // Auto-create sheet with headers if it doesn't exist
    if (!sheet) {
      sheet = ss.insertSheet('Orders');
      sheet.appendRow([
        'Order ID', 'Date & Time', 'Service',
        'Instagram ID / Link', 'Quantity',
        'Amount (₹)', 'Promo Code', 'Status'
      ]);
      // Style the header row
      var header = sheet.getRange(1, 1, 1, 8);
      header.setFontWeight('bold');
      header.setBackground('#f09433');
      header.setFontColor('#ffffff');
    }

    var data = JSON.parse(e.postData.contents);

    sheet.appendRow([
      data.orderId,
      data.timestamp,
      data.service,
      data.target,
      data.quantity,
      data.amount,
      data.promo,
      data.status
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({ result: 'ok' }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ result: 'error', msg: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
